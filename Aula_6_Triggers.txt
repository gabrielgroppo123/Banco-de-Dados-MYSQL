CREATE Trigger actor_copia after update on actor for each row
begin
declare nome varchar(50);
IF new.first_name <> old.first_name then
  set nome  = CONCAT(OLD.first_name, " ", OLD.last_name);
  insert into copia_actor (actor_id,nome, usuario, data_alteracao) VALUES(old.actor_id, nome, "Guilherme", NOW());
end if;
end;

drop TRIGGER actor_copia;
